Задание 1
select c.login as courier_login, count(o.id) FILTER(where o."inDelivery" = true) as orders_Count from "Couriers" as c LEFT JOIN "Orders" as o ON c.id = o."courierId" GROUP BY c.login;
Задание 2
select o.track, CASE WHEN o.finished = true THEN 2 WHEN o.cancelled = true THEN -1 WHEN o."inDelivery" THEN 1 ELSE 0 END as Status from "Orders" as o;
